select distinct  app."ORDER", tiu."ORDER", prod."ID", it."NAME", it."LABEL", val."ITEM_VALUE"
from "VALUE" val
inner join "CHARACTERISTIC_VALUE" cval on val."CHARACTERISTIC_VALUE_ID" = cval."ID"
inner join "PRODUCT" prod on cval."PRODUCT_ID" = prod."ID"
inner join "PRODUCT_RANGE" rang on prod."PRODUCT_RANGE_ID" = rang."ID"
inner join "TECHNICAL_ITEM_USE" tiu on cval."TECHNICAL_ITEM_USE_ID" = tiu."ID"
inner join "CHARACTERISTIC_ITEM" it on tiu."CHARACTERISTIC_ITEM_ID" = it."ID"

inner join "TECHNICAL_DEFINITION" td on tiu."TECHNICAL_DEFINITION_ID" = td."ID"
inner join "APPLICATION" app on td."APPLICATION_ID" = app."ID"
inner join "technical_definition_external_export_data" tdeed on tdeed."TECHNICAL_DEFINITION_ID" = td."ID"
inner join "EXTERNAL_EXPORT_DATA" eed on  eed."ID" = tdeed."EXTERNAL_EXPORT_DATA_ID"

inner join "PROGRAM" prog on app."PROGRAM_ID" = prog."ID"
inner join "CONTRACT" ctr on ctr."PROGRAM_ID" = prog."ID"
inner join "CAMPAIGN" camp on camp."CONTRACT_ID" = ctr."ID"
inner join "REFERENCE_YEAR" year on camp."REFERENCE_YEAR_ID" = year."ID"
inner join "CONTRACT_EXTERNAL_EXPORT_DATA" ctreed on ctreed."CONTRACT_ID" = ctr."ID"
inner join "EXTERNAL_EXPORT_DATA" eedCtr on  eedCtr."ID" = ctreed."EXTERNAL_EXPORT_DATA_ID"
where 
year."YEAR" = 2018 and
ctr."ID" = 1846 and 
eedCtr."CODE" = 'BAFA' and 
prod."ID" in (5519870, 5519869, 5519872, 5519873, 5519871) and 
rang."ID" = 52244
order by app."ORDER" asc, tiu."ORDER" asc
