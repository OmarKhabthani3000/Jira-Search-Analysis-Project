create table customer (
id number primary key,
first_name varchar2(50),
last_name varchar2(50)
)
/

create table customer_contact(
customer_id number,
serial_id number,
phone varchar2(50)
)
/

alter table customer_contact add constraint pk_customer_contact primary key (customer_id, serial_id)
/

alter table customer_contact add constraint fk_customer_contact foreign key (customer_id) references customer(id)
/
