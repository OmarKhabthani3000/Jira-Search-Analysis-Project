	private void importScript(Reader importFileReader, Statement statement)
			throws IOException {
		log.info( "Executing import script: " + importFile );
		StringBuilder trimmedSql = new StringBuilder();
		BufferedReader reader = new BufferedReader( importFileReader );
		try {
			for (String sql = reader.readLine(); sql != null; sql = reader.readLine()) {
				String trimmedSqlLine = sql.trim();
				if (0 == trimmedSqlLine.length() ||
				     trimmedSqlLine.startsWith("--") ||
				     trimmedSqlLine.startsWith("//") ||
				     (trimmedSqlLine.startsWith("/*") && trimmedSqlLine.endsWith("*/;"))) {
					continue;
				}
				else {
					if (trimmedSqlLine.endsWith(";")) {
						trimmedSql.append(" ").append(trimmedSqlLine.substring(0, trimmedSqlLine.length() - 1));
						log.debug(trimmedSql);
						statement.execute(trimmedSql.toString());
						trimmedSql.setLength(0);
					}
					else {
						trimmedSql.append(trimmedSqlLine);
					}
				}
			}
			if (trimmedSql.length() > 0) {
				log.debug(trimmedSql);
				statement.execute(trimmedSql.toString());
				trimmedSql.setLength(0);
			}
		}
		catch (SQLException e) {
			throw new JDBCException("Error during import script execution", e);
		}
	}
